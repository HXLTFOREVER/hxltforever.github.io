23.工具提示

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <link rel="stylesheet" href="css/bootstrap.min.css" />
    <script src="js/jquery.min.js"></script>
    <script src="js/bootstrap.min.js"></script>
</head>
<body>
    <!--
        对于工具提示， data-toggle="tooltip"
        title: 提示框中的信息。
        data-placement： 控制提示框出现的方向。
     -->
    <button class="btn btn-default" data-toggle="tooltip" title="hello world" data-placement="right">按钮</button>

    <button class="btn btn-default" data-toggle="popover" data-placement="right" title="提示" data-content="这是一个提示信息">按钮</button>

</body>
<script>
    // 处于对bootstrap性能的原因，这两个功能是选择性接入的(bootstrap没有强制性将这两个功能初始化)，
    // 所以我们要使用的时候必须要初始化。
    $('button[data-toggle="tooltip"]').tooltip();  //

    $('button[data-toggle="popover"]').popover();  //
</script>
</html>

25.按钮的状态

<body>
    <!-- data-loading-text当按钮不可点的时候显示文字。 -->
    <button id="btn" class="btn btn-primary" data-loading-text="提交中..." >提交</button>

<!--    <input type="checkbox" id="interests"> -->
</body>
<script>
    /**  效果和下面是一样的
    $('#btn').click(function() {})
     */
    /**  */
    $('#btn').on('click', function() {
        // button('loading') 表示按钮处于加载的过程中, loading是固定值
        $('#btn').button('loading');

        setTimeout(function() {
            // 让按钮恢复原始的状态, reset是固定
            $('#btn').button('reset');
        }, 2000);
    })

    /**
    $('#interests').on('change', function() {

    });
     */

</script>
</html>

26.折叠框

<body>
    <!--
        控制折叠, 可以使用a和button标签。如果是a标签用 href; 如果是button是data-target，
        但是两种方式都必须加上 data-toggle="collapse"
     -->
    <a class="btn btn-danger" href="#exampleCollapse" data-toggle="collapse">折叠</a>

    <button class="btn btn-danger" data-target="#exampleCollapse" data-toggle="collapse">折叠</button>


    <div class="collapse" id="exampleCollapse">
        这是一个折叠信息
    </div>
</body>

27.轮播图

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <link rel="stylesheet" href="css/bootstrap.min.css" />
    <link rel="stylesheet" href="css/font-awesome.css" />
    <script src="js/jquery.min.js"></script>
    <script src="js/bootstrap.min.js"></script>
    <style>
        .control-icon {
            position: absolute;
            top: 50%;
        }
    </style>
</head>
<body>

    <div id="carousel-example" style="margin: 0 auto; width: 590px; height: 470px;" class="carousel slide" data-ride="carousel">
        <!-- 小圆点 -->
        <ol class="carousel-indicators">
            <li data-target="#carousel-example" data-slide-to="0" class="active"></li>
            <li data-target="#carousel-example" data-slide-to="1"></li>
            <li data-target="#carousel-example" data-slide-to="2"></li>
        </ol>

        <!-- 放置图片 -->
        <div class="carousel-inner">

            <!-- 展示每张图片以及文字 -->
            <div class="item active">
                <img width="590px" height="470px" src="images/image1.jpg">

            </div>
            <div class="item">
                <img width="590px" height="470px" src="images/image2.jpg">
                <div class="carousel-caption">
                    <h3>图片的描述信息</h3>
                    <p>展示每张图片以及文字 展示每张图片以及文字 展示每张图片 </p>
                </div>
            </div>
            <div class="item">
                <img width="590px" height="470px" src="images/image3.jpg">
            </div>
        </div>

        <!-- 左边的箭头 -->
        <a class="left carousel-control" href="#carousel-example" data-slide="prev">
            <span class="fa fa-chevron-left control-icon"></span>
            <span class="sr-only">Previous</span>
        </a>
        <!-- 右边的箭头 -->
        <a class="right carousel-control" href="#carousel-example" data-slide="next">
            <span class="fa fa-chevron-right control-icon"></span>
            <span class="sr-only">Next</span>
        </a>
    </div>
</body>
</html>
